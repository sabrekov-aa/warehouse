SELECT "Название роли",sum("Количество матчей") AS "Сумма матчей" FROM (SELECT * FROM (SELECT * FROM "Профессиональные игроки" AS "Пр_игр" INNER JOIN "Игрок_Роль" ON ("Пр_игр"."Код игрока" = "Игрок_Роль"."Код игрока")) AS 
"Промеж_этап1" INNER JOIN "Роли в игре" ON "Промеж_этап1"."Код роли"="Роли в игре"."Код роли" WHERE "Процент побед">60) AS "Промеж_этап2" GROUP BY "Промеж_этап2"."Название роли"
Показывает сколько игр сыгранно на каждой роли профессиональными игроками, причем в расчет берутся только игры особо результативных игроков, чей процент побед более 60.

SELECT "Характеристика", avg("Процент побед") AS "Средний процент побед" FROM (SELECT * FROM (SELECT * FROM "Герои" INNER JOIN "Герой_Роль" ON ("Герои"."Код героя" = "Герой_Роль"."Код Героя") WHERE "Частота выбора">16) AS "Промеж_этап" INNER JOIN "Роли в игре" ON "Промеж_этап"."Код Роли" = "Роли в игре"."Код роли"  WHERE "Название роли" = 'Керри') AS "Табл_без_хар" INNER JOIN "Характеристики героев" ON "Основная характеристика героя"="Код характеристики" GROUP BY "Характеристика"
Показывает средний процент побед популярных героев (регулируется через частоту выбора) определенной роли (в данном случае керри) с каждой характеристикой героя.
