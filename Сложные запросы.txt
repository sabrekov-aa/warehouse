1)SELECT "Название команды", "Ник игрока", "Имя", "Фамилия", "Название страны" FROM (SELECT * FROM "Команды" AS "К" INNER JOIN "Профессиональные игроки" AS "Пр_иг" ON "Команда игрока"="Код команды") AS "Инф_без_стран" INNER JOIN "Страны" ON "Страна" = "Код страны" ORDER BY "Популярность_место" (Выдает таблицу с данными игроков, сгруппированные по командам, которые отсортированы в порядке популярности)

2)SELECT "Имя героя", "Процент побед" FROM (SELECT * FROM (SELECT * FROM "Герои" INNER JOIN "Герой_Роль" ON ("Герои"."Код героя" = "Герой_Роль"."Код Героя")) AS "Промеж_этап" INNER JOIN "Роли в игре" ON "Промеж_этап"."Код Роли" = "Роли в игре"."Код роли") AS "Табл_без_хар" INNER JOIN "Характеристики героев" ON "Основная характеристика героя"="Код характеристики" WHERE ("Характеристика" = 'Ловкость' AND "Название роли" = 'Керри') ORDER BY "Табл_без_хар"."Процент побед" DESC (Показывает героев, роль которых керри, а основная характеристика ловкость, сортирует в порядке убывания процента побед)
Оптимизация: Был добавлен индекс "хар_гер_инд".CREATE INDEX "хар_гер_инд" ON "Характеристики героев"("Характеристика");

3)SELECT "Ник игрока","Количество матчей"
FROM (SELECT * FROM (SELECT * FROM "Профессиональные игроки" AS "Пр_игр" INNER JOIN "Игрок_Роль" ON ("Пр_игр"."Код игрока" = "Игрок_Роль"."Код игрока")) AS 
"Промеж_этап1" INNER JOIN "Роли в игре" ON "Промеж_этап1"."Код роли"="Роли в игре"."Код роли") AS "Табл_без_стран" INNER JOIN "Страны" ON 
"Табл_без_стран"."Страна"="Страны"."Код страны" WHERE ("Процент побед">50 AND "Название роли" = 'Керри' AND "Название страны" = 'Шведция') ORDER BY "Количество матчей" DESC (Выводит таблицу керри из определенной страны с процентом побед более 50, сортирует их по кол-ву матчей)
Оптимизация: был добавлен индекс "страны_инд"
CREATE INDEX "cтраны_инд" ON "Страны"("Название страны");
