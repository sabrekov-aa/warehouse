1)SELECT "Ник игрока", "Название страны" FROM (SELECT "Код игрока","Ник игрока", "Страна" FROM "Профессиональные игроки") AS "Код_и_страна" INNER JOIN "Страны" ON "Страна"="Код страны"; (Получаем таблицу с ником игрока и с названием страны, из которой он)
2)SELECT "Ник игрока", "Процент побед к"+"Процент побед и" AS "Сумма_поб" FROM (SELECT "Код команды","Процент побед" AS "Процент побед к" FROM "Команды") AS "Проц_поб_ком" INNER JOIN
(SELECT "Код игрока","Ник игрока","Команда игрока","Процент побед" AS "Процент побед и" FROM "Профессиональные игроки") AS "Общество_победителей" ON "Команда игрока"="Код команды" ORDER BY "Сумма_поб" DESC; (Показывает таблицу, в которой в левом столбце ник игрока, а справа сумма его процента побед и процента побед его команды (в порядке убывания))(Таблица победителей по жизни)
3)SELECT "Имя героя", "Соотношение УСП" FROM(SELECT "Имя героя", "Основная характеристика героя","Соотношение УСП","Процент побед" FROM "Герои"  ORDER BY "Процент побед" > 50) AS "Гер_хар" RIGHT JOIN (SELECT * FROM "Характеристики героев" WHERE "Код характеристики" > 2) AS "Хар_хар" ON "Основная характеристика героя" = "Код характеристики"; (Показывает таблицу с с именами героев (с основным атрибутом интеллект и процентом побед более 50) и их соотношение УСП)
